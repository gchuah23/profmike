<link rel="stylesheet" type="text/css" href="https://cdnjs.cloudflare.com/ajax/libs/slick-carousel/1.8.1/slick.css"/>

<link rel="stylesheet" type="text/css" href="https://cdnjs.cloudflare.com/ajax/libs/slick-carousel/1.8.1/slick-theme.min.css"/>

 

<script>

AOS.init({

once: true, // Only run animations once

});

 

$(document).ready(function() {

// Testimonial Slider

$('.testimonial-slider').slick({

dots: true,

infinite: true,

speed: 500,

slidesToShow: 1,

adaptiveHeight: true,

autoplay: true,

autoplaySpeed: 5000,

});

 

// Smooth scrolling for navigation links

$('a[href^="#"]').on('click', function(e) {

e.preventDefault();

 

var target = this.hash;

var $target = $(target);

 

$('html, body').animate({

'scrollTop': $target.offset().top

}, 800, 'swing');

});

 

// Highlight active navigation link on scroll

$(document).on('scroll', function() {

var scrollPosition = $(this).scrollTop();

$('nav a').each(function() {

var currentLink = $(this);

var refElement = $(currentLink.attr("href"));

if (refElement.position().top <= scrollPosition && refElement.position().top + refElement.height() > scrollPosition) {

$('nav a').removeClass("active");

currentLink.addClass("active");

} else {

currentLink.removeClass("active");

}

});

});

 

// Sticky navbar

$(window).on('scroll', function() {

if ($(window).scrollTop() > 100) {

$('#navbar').addClass('shadow-lg');

$('#navbar').removeClass('py-4');

$('#navbar').addClass('py-3');

} else {

$('#navbar').removeClass('shadow-lg');

$('#navbar').addClass('py-4');

$('#navbar').removeClass('py-3');

}

});

 

// Mobile menu toggle

$('#hamburger-btn').on('click', function() {

$('#mobile-menu').removeClass('hidden');

$('body').addClass('overflow-hidden');

});

 

$('#close-menu-btn').on('click', function() {

$('#mobile-menu').addClass('hidden');

$('body').removeClass('overflow-hidden');

});

 

$('.mobile-nav-link').on('click', function() {

$('#mobile-menu').addClass('hidden');

$('body').removeClass('overflow-hidden');

});

 

// Form submission (basic example)

$('#contact-form').on('submit', function(e) {

e.preventDefault();

var name = $('#name').val();

var email = $('#email').val();

var message = $('#message').val();

 

// Basic validation (you should use a more robust validation library)

if (name === '' || email === '' || message === '') {

alert('Please fill in all fields.');

return;

}

 

// In a real application, you would send the data to a server using AJAX

console.log('Name:', name);

console.log('Email:', email);

console.log('Message:', message);

alert('Message sent successfully! (This is a demo)');

$('#contact-form')[0].reset(); // Clear the form

});

// Auto-highlight active nav item

$(window).on('scroll', function() {

var currentSectionId = '';

$('section').each(function(){

var top = $(this).offset().top - 100; // Adjust as needed

var id = $(this).attr('id');

if ($(window).scrollTop() >= top) {

currentSectionId = id;

}

});

$('nav .nav-link').removeClass('active');

$('nav a[href="#' + currentSectionId + '"]').addClass('active');

 

$('.mobile-nav-link').removeClass('active');

$('.mobile-nav-link[href="#' + currentSectionId + '"]').addClass('active');

});

});

</script>

</body>

</html>
